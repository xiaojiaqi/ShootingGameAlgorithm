List2-9 向16個方向射出，速度為3的子彈
void InitDirectedBullet16_3(
	float& vx, float& vy,  //子彈的速度
	float theta            //發射的角度
) {
	//速度3的速度表單
	static int v3[][2]={
		{ 3, 0}, { 3, 1}, { 2, 2}, { 1, 3},
		{ 0, 3}, {-1, 3}, {-2, 2}, {-3, 1},
		{-3, 0}, {-3,-1}, {-2,-2}, {-1,-3},
		{ 0,-3}, { 1,-3}, { 2,-2}, { 3,-1}
	};

	//把角度theta（0°∼360°）變換成16個方向（0∼15）
	int dir=theta*16/360;

	//由表單求出子彈速度(vx,vy) 
	vx=v3[dir][0];
	vy=v3[dir][1];
}
